安装升级
列表
curl --request POST \
  --url http://localhost:18006/lpnode/lpnode_admin_panel/ctrl/listInstall \
  --header 'Content-Type: application/json' \
  --data '{
	"installType":"ammClient"
}'

安装lpclient
curl --request POST \
  --url http://localhost:18006/lpnode/lpnode_admin_panel/ctrl/installLpClient \
  --header 'Content-Type: application/json' \
  --data '{
	"setupConfig": {
		"type": "near",
		"imageRepository": "kldtks/edge:obridge-chain-client-near-37b587b2",
		"install":true,
		"startBlock":"100"
	}
}'

卸载lpclient
curl --request POST \
  --url http://localhost:18006/lpnode/lpnode_admin_panel/ctrl/uninstallLpClient \
  --header 'Content-Type: application/json' \
  --data '{
	"setupConfig": {
		"type": "near",
		"uninstall":true
	}
}'
